# CLI Stack

The corpo CLI is a TypeScript binary built with incur, compiled with Bun, and
distributed as a single self-contained executable. The GUI SPA is embedded
in the binary at compile time.

See [GUI Stack](b5f3e2d1a4c84f9ab1e7c2d6f3b5a4e8) for the GUI side.

---

## Runtime — Bun

Bun compiles TypeScript to a single binary with no external runtime dependency:

```sh
bun build --compile --minify src/index.ts --outfile corpo
```

Bun is chosen for three reasons specific to corpo:

1. **Asset embedding** — the GUI `dist/` directory is embedded into the
   binary at compile time. Bun handles this natively via `--asset-naming`.
2. **Built-in SQLite** — the cache DB (`~/.corpo/cache.db`) uses
   `bun:sqlite` with no additional dependency.
3. **Built-in HTTP server** — `Bun.serve()` powers `corpo serve` with no
   extra install.

## HTTP Layer — Hono

`corpo serve` exposes the six-endpoint API and serves the embedded GUI.
[Hono](https://hono.dev) handles routing (~14KB, runs on Bun natively):

```ts
import { Hono } from 'hono'
import { serveStatic } from 'hono/bun'

export function createServer(root: string) {
  const app = new Hono()

  app.get('/api/navigation', async (c) => {
    const config = await readConfig(root)
    return c.json(config)
  })

  app.put('/api/navigation', async (c) => {
    const { navigation } = await c.req.json()
    await writeConfig(root, { ...await readConfig(root), navigation })
    return c.body(null, 204)
  })

  app.get('/api/files/:id', async (c) => {
    const raw = await readFile(root, c.req.param('id'))
    if (raw === null) return c.notFound()
    return c.text(raw)
  })

  app.post('/api/files', async (c) => {
    const { raw } = await c.req.json()
    const id = randomUUID().replace(/-/g, '')
    await writeCorpoFile(root, id, raw)
    return c.json({ id }, 201)
  })

  app.put('/api/files/:id/commit', async (c) => {
    const id = c.req.param('id')
    if (!await fileExists(root, id)) return c.notFound()
    const { raw } = await c.req.json()
    await writeCorpoFile(root, id, raw)
    return c.body(null, 204)
  })

  app.delete('/api/files/:id', async (c) => {
    const id = c.req.param('id')
    await deleteFile(root, id)  // handles local + remote + navigation cleanup
    return c.body(null, 204)
  })

  // Serve embedded GUI SPA for all other routes
  app.get('*', serveStatic({ root: './gui/dist' }))

  return app
}
```

## Shared Primitives

### Finding the root

Every command walks up from CWD, mirroring how git finds `.git/`:

```ts
import { access } from 'node:fs/promises'
import { dirname, join } from 'node:path'

export async function findRoot(start = process.cwd()): Promise<string> {
  let dir = start
  while (true) {
    try {
      await access(join(dir, '.corpo', 'config.json'))
      return dir
    } catch {
      const parent = dirname(dir)
      if (parent === dir) throw new Error('Not inside a corpo directory')
      dir = parent
    }
  }
}
```

### Generating file IDs

```ts
import { randomUUID } from 'node:crypto'

export const generateId = () => randomUUID().replace(/-/g, '')
```

### Reading and writing config

```ts
import { readFile, writeFile } from 'node:fs/promises'
import { join } from 'node:path'

export async function readConfig(root: string) {
  const raw = await readFile(join(root, '.corpo', 'config.json'), 'utf8')
  return JSON.parse(raw)
}

export async function writeConfig(root: string, config: unknown) {
  await writeFile(
    join(root, '.corpo', 'config.json'),
    JSON.stringify(config, null, 2),
  )
}
```

## Command Implementations

### `corpo init`

```ts
import { mkdir, writeFile, access } from 'node:fs/promises'
import { resolve, join } from 'node:path'

export const init = {
  description: 'Scaffold a new Corpo-enabled repository.',
  args: z.object({
    path: z.string().optional().describe('Target directory (defaults to CWD)'),
  }),
  async run(c) {
    const root = resolve(c.args.path ?? '.')
    const corpoDir = join(root, '.corpo')

    try {
      await access(corpoDir)
      throw new Error('.corpo/ already exists')
    } catch (e: any) {
      if (e.code !== 'ENOENT') throw e
    }

    await mkdir(join(corpoDir, 'files'), { recursive: true })
    await writeFile(
      join(corpoDir, 'config.json'),
      JSON.stringify({ navigation: [] }, null, 2),
    )

    return { root, created: ['.corpo/', '.corpo/files/', '.corpo/config.json'] }
  },
}
```

### `corpo new`

```ts
import { writeFile } from 'node:fs/promises'
import { join } from 'node:path'

export const new_ = {
  description: 'Create a new file in the current directory.',
  options: z.object({
    title: z.string().describe('File title'),
    description: z.string().describe('One-line file description'),
  }),
  async run(c) {
    const root = await findRoot()
    const id = generateId()
    const raw = [
      '---',
      `title: "${c.options.title}"`,
      `description: "${c.options.description}"`,
      '---',
      '',
    ].join('\n')

    await writeFile(join(root, '.corpo', 'files', `${id}.md`), raw)

    return c.ok({ id, path: `.corpo/files/${id}.md` }, {
      cta: { commands: [{ command: 'serve', description: 'Open in GUI' }] },
    })
  },
}
```

### `corpo serve`

```ts
import { createServer } from '../server/index.ts'

export const serve = {
  description: 'Start the local GUI server.',
  options: z.object({
    port: z.number().default(3000).describe('Port to listen on'),
  }),
  outputPolicy: 'agent-only',
  async run(c) {
    const root = await findRoot()
    const app = createServer(root)

    const server = Bun.serve({ fetch: app.fetch, port: c.options.port })

    if (!c.agent) {
      console.log(`GUI running at http://localhost:${server.port}`)
      Bun.openInEditor(`http://localhost:${server.port}`)  // opens browser
    }

    // Keep alive until Ctrl+C
    await new Promise(() => {})

    return { url: `http://localhost:${server.port}` }
  },
}
```

### `corpo sync`

```ts
export const sync = {
  description: 'Pull latest content from all configured remotes.',
  options: z.object({
    remote: z.string().optional().describe('Sync a specific remote only'),
  }),
  async run(c) {
    const root = await findRoot()
    const config = await readConfig(root)
    const remotes = c.options.remote
      ? filterByRemote(config.remote_files, c.options.remote)
      : config.remote_files

    const results = await syncRemotes(remotes)
    return { updated: results.updated, unchanged: results.unchanged, new: results.new }
  },
}
```

### `corpo publish`

```ts
export const publish = {
  description: 'Register a file with corpo.sh to get a persistent link.',
  args: z.object({
    id: z.string().describe('File ID to publish'),
  }),
  async run(c) {
    const root = await findRoot()
    const filePath = join(root, '.corpo', 'files', `${c.args.id}.md`)

    await access(filePath).catch(() => {
      throw new Error(`File not found: ${c.args.id}`)
    })

    const res = await fetch('https://corpo.sh/api/files', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id: c.args.id }),
    })

    if (!res.ok) throw new Error(`corpo.sh responded ${res.status}`)

    return { url: `https://corpo.sh/${c.args.id}` }
  },
}
```

## Entry Point

```ts
// src/index.ts
import { Cli, z } from 'incur'
import { init } from './commands/init.ts'
import { new_ } from './commands/new.ts'
import { serve } from './commands/serve.ts'
import { sync } from './commands/sync.ts'
import { publish } from './commands/publish.ts'
import { auth } from './commands/auth/index.ts'

Cli.create('corpo', {
  description: 'Git-backed, agent-native file protocol.',
})
  .command('init', init)
  .command('new', new_)
  .command('serve', serve)
  .command('sync', sync)
  .command('publish', publish)
  .command(auth)
  .serve()
```
